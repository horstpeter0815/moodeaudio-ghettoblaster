# Copy and paste these commands to setup WiFi:

# Option 1: If bootfs is mounted at /Volumes/bootfs
sudo tee /Volumes/bootfs/wpa_supplicant.conf > /dev/null << 'WIFIEOF'
country=US
ctrl_interface=DIR=/var/run/wpa_supplicant GROUP=netdev
update_config=1

network={
    ssid="Centara Nova Hotel"
    key_mgmt=WPA-EAP
    eap=PEAP
    identity="309"
    password="Password"
    phase2="auth=MSCHAPV2"
}
WIFIEOF

# Option 2: If boot partition is in rootfs/boot
sudo tee /Volumes/rootfs/boot/wpa_supplicant.conf > /dev/null << 'WIFIEOF'
country=US
ctrl_interface=DIR=/var/run/wpa_supplicant GROUP=netdev
update_config=1

network={
    ssid="Centara Nova Hotel"
    key_mgmt=WPA-EAP
    eap=PEAP
    identity="309"
    password="Password"
    phase2="auth=MSCHAPV2"
}
WIFIEOF

# Verify
cat /Volumes/bootfs/wpa_supplicant.conf 2>/dev/null || cat /Volumes/rootfs/boot/wpa_supplicant.conf
