# Run this command to create WiFi config:

sudo tee /Volumes/bootfs/wpa_supplicant.conf > /dev/null << 'WIFIEOF'
country=US
ctrl_interface=DIR=/var/run/wpa_supplicant GROUP=netdev
update_config=1

network={
    ssid="Centara Nova Hotel"
    key_mgmt=WPA-EAP
    eap=PEAP
    identity="309"
    password="Password"
    phase2="auth=MSCHAPV2"
}
WIFIEOF

# Then verify:
cat /Volumes/bootfs/wpa_supplicant.conf
